{Soma de matrizes}
var a, b, c: array[1..3, 1..3] of integer;
i, j, k: integer;

begin
    {Leitura das matrizes}

    for i := 1 to 3 do

        for j := 1 to 3 do begin

        end;

        writeln;
    end;
end.
