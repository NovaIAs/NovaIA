```
.MODEL SMALL
.STACK 100H
.DATA
	STRING1 DB "FROG", 13, 10, 0DH
	STRING2 DB "TOAD", 13, 10, 0DH
	STRING3 DB "NEWT", 13, 10, 0DH
	STRING4 DB "SALAMANDER", 13, 10, 0DH
	 BUFFER DB 20 DUP (?) ; DOUBLE NULL TERMINATED
.CODE
MAIN PROC FAR
	MOV AX, @DATA
	MOV DS, AX
	MOV SI, OFFSET STRING1
	MOV DX, OFFSET BUFFER
	CALL PRINT_STRING
	MOV SI, OFFSET STRING2
	CALL PRINT_STRING
	MOV SI, OFFSET STRING3
	CALL PRINT_STRING
	MOV SI, OFFSET STRING4
	CALL PRINT_STRING
	MOV AX, 4C00H
	INT 21H
MAIN ENDP
PRINT_STRING PROC FAR
	PUSH SI
	PUSH DI
	LODSB
	CMP AL, 0
	JE FINISH
	MOV AH, 02H
	MOV DL, AL
	INT 21H
	JMP PRINT_STRING
FINISH:
	POP DI
	POP SI
	RET
PRINT_STRING ENDP
END MAIN
```
