```shell
#!/bin/bash
#
# This script generates a random password of a given length.
#
# Usage:
#   generate_password.sh <length>
#
# Example:
#   generate_password.sh 12
#

# Check if the user has provided a length.
if [ $# -ne 1 ]; then
  echo "Usage: generate_password.sh <length>"
  exit 1
fi

# Get the length of the password.
length=$1

# Define the characters that can be used in the password.
characters="abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$%^&*()_+=-"

# Generate the password.
password=""
for (( i=0; i<$length; i++ )); do
  # Get a random character from the list of characters.
  character=$(echo $characters | fold -w1 | shuf | head -n1)

  # Add the character to the password.
  password=$password$character
done

# Print the password.
echo $password
```

The script then defines the characters that can be used in the password and generates a random password of the specified length. The password is then printed to the console.
