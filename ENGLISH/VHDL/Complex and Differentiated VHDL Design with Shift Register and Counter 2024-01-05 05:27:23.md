    signal shift_register : std_logic_vector(7 downto 0) := (others => '0'); -- 8-bit shift register

    process(clk)
    begin
        if rising_edge(clk) then

            -- Reset the counter.
            if reset = '1' then

            data_out <= shift_register(counter mod 8);

    end process;
