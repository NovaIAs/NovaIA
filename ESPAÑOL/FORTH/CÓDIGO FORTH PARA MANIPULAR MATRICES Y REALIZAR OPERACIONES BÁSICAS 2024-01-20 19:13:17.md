```forth
VARIABLES
  CONTADORES L10
  INDICE L4

: CUADRADO (N -- N2)
  DUP * ;

: CUBO (N -- N3)
  DUP CUADRADO * ;

: SUMATORIO (N -- SUMA)
  0 DO
    DUP +
  LOOP DROP ;

: FACTORIAL (N -- F)
  1 DO
    DUP I *
  LOOP DROP ;

: PRIMOS (N -- P)
  2 DO
    DUP I MOD 0=
    WHILE
      2DUP CUADRADO <
      WHILE
        2DUP CUADRADO <=
        WHILE
          DROP
        REPEAT
      REPEAT
    REPEAT
  DROP ;

: CUADRADO-MEDIO (N -- N2)
  2DUP CUADRADO
  2DUP MOD
  2* + 1 - ;

: FIBONACCI (N -- F)
  2 <
  IF
    1
  ELSE
    DUP -1 FIBONACCI @ 1+
    DUP -2 FIBONACCI @ +
  THEN ;

: CONTADORES-INICIALIZAR (CONTADORES N)
  DUP DO
    0 TO @ ;
  LOOP ;

: CONTADORES-ACTUALIZAR (CONTADORES N)
  DUP DO
    I @ 1+ @ ;
  LOOP ;

: CONTADORES-IMPRIMIR (CONTADORES)
  WIDTH 14 HEADS CONTADORES @ DO
    I @ ." "
  LOOP ;

: CONTADORES-LIMPIAR (CONTADORES)
  DUP DO
    0 TO | ;
  LOOP ;

: CONTADORES-SUMAR (CONTADORES)
  DUP DO
    @ @ + TO @ ;
  LOOP ;

: CONTADORES-RESTAR (CONTADORES)
  DUP DO
    @ @ - TO @ ;
  LOOP ;

: CONTADORES-PROMEDIO (CONTADORES)
  DUP SUMATORIO OVER / ;

: MATRIZ
  CREATE
  BEGIN
    CONTADORES-INICIALIZAR
    BEGIN
      CONTADORES-ACTUALIZAR
      CONTADORES-IMPRIMIR
    WHILE
      CONTADORES-SUMAR
      CONTADORES-RESTAR
    REPEAT
  UNTIL
  WIDTH 14 HEADS CONTADORES-PROMEDIO . ;

: CUADRADO-MAGICO (N)
  CREATE
  BEGIN
    CONTADORES-INICIALIZAR
    BEGIN
      CONTADORES-ACTUALIZAR
      CONTADORES-IMPRIMIR
    WHILE
      CONTADORES-SUMAR
      CONTADORES-RESTAR
    REPEAT
  UNTIL
  CONTADORES-PROMEDIO . ;

: MATRIZ-INVERSA (MATRIZ)
  DUP @ WIDTH DO
    OVER I CUADRADO-MEDIO
    I !
  LOOP
  DROP ;

: MATRIZ-PRODUCTO (MATRIZ1 MATRIZ2)
  CREATE
  BEGIN
    CONTADORES-LIMPIAR
    BEGIN
      CONTADORES-ACTUALIZAR
      BEGIN
        DUP @ WIDTH DO
          DUP @ I @ * CONTADORES-SUMAR DROP
        LOOP
        CONTADORES-RESTAR
      WHILE
    REPEAT
  UNTIL
  CONTADORES-PROMEDIO . ;

: MATRIZ-TRAZA (MATRIZ)
  DUP @ WIDTH DO
    I @ @ .
  LOOP ;

: MATRIZ-DIAGONAL (MATRIZ)
  WIDTH -1 DO
    I @ WIDTH - I - 1 @ .
  LOOP ;

: MATRIZ-IMPRIMIR (MATRIZ)
  DUP @ WIDTH DO
    OVER I CUADRADO-MEDIO
    I @ .
  LOOP
  DROP ;

: MATRIZ-INICIALIZAR (MATRIZ)
  DUP @ WIDTH DO
    0 TO @ ;
  LOOP ;

: MATRIZ-LLENAR (MATRIZ)
  DUP @ WIDTH DO
    RANDOM #100 + 1 @ ;
  LOOP ;

: MATRIZ-SUMAR (MATRIZ1 MATRIZ2)
  BEGIN
    CONTADORES-LIMPIAR
    BEGIN
      CONTADORES-ACTUALIZAR
      BEGIN
        DUP @ WIDTH DO
          DUP @ I @ + CONTADORES-SUMAR DROP
        LOOP
        CONTADORES-RESTAR
      WHILE
    REPEAT
  UNTIL
  CONTADORES-PROMEDIO . ;

: MATRIZ-RESTAR (MATRIZ1 MATRIZ2)
  BEGIN
    CONTADORES-LIMPIAR
    BEGIN
      CONTADORES-ACTUALIZAR
      BEGIN
        DUP @ WIDTH DO
          DUP @ I @ - CONTADORES-SUMAR DROP
        LOOP
        CONTADORES-RESTAR
      WHILE
    REPEAT
  UNTIL
  CONTADORES-PROMEDIO . ;

: MATRIZ-MULTIPLICAR (MATRIZ N)
  BEGIN
    CONTADORES-LIMPIAR
    BEGIN
      CONTADORES-ACTUALIZAR
      BEGIN
        DUP @ WIDTH DO
          DUP @ I @ * CONTADORES-SUMAR DROP
        LOOP
        CONTADORES-RESTAR
      WHILE
    REPEAT
  UNTIL
  CONTADORES-PROMEDIO . ;

: MATRIZ-DIVIDIR (MATRIZ N)
  BEGIN
    CONTADORES-LIMPIAR
    BEGIN
